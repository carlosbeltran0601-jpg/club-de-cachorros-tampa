// REPLACE your current savePuppy function with this one: async function savePuppy() { const breed = document.getElementById('pBreed').value; const price = document.getElementById('pPrice').value; const locationInput = document.getElementById('pLocation').value; const weeksOld = parseInt(document.getElementById('pAge').value); // 1. Get ALL files selected const fileInput = document.getElementById('pFile'); const files = fileInput.files; if(!breed || !price || files.length === 0) { alert("Please fill breed, price, and select at least one photo."); return; } const btn = document.getElementById('postBtn'); btn.innerText = `Uploading ${files.length} Photos...`; btn.disabled = true; try { const imageUrls = []; // 2. Loop through every file and upload individually for (let i = 0; i < files.length; i++) { const file = files[i]; const storageRef = storage.ref('puppies/' + Date.now() + "_" + file.name); const snapshot = await storageRef.put(file); const downloadURL = await snapshot.ref.getDownloadURL(); imageUrls.push(downloadURL); // Collect the URLs in an array } // 3. Calculate BirthDate const birthDate = new Date(); birthDate.setDate(birthDate.getDate() - (weeksOld * 7)); // 4. Save to Firestore await db.collection("puppies").add({ breed: breed, price: price, location: locationInput || "Tampa, FL", birthDate: firebase.firestore.Timestamp.fromDate(birthDate), images: imageUrls, // Store as an array 'images' image: imageUrls[0], // Store first one as 'image' for backwards compatibility healthCert: document.getElementById('pHealth').checked, dewormed: document.getElementById('pDewormed').checked, isVerified: document.getElementById('pVerified').checked, sellerEmail: currentUserEmail, createdAt: firebase.firestore.FieldValue.serverTimestamp() }); alert("Puppy posted successfully with " + files.length + " photos!"); closeModal(); location.reload(); } catch (e) { alert("Upload Error: " + e.message); btn.innerText = "Publish Listing"; btn.disabled = false; } }
