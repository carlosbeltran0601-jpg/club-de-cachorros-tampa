// Initialize Firebase firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); const signinForm = document.getElementById('signinForm'); const loginBtn = document.getElementById('loginBtn'); signinForm.onsubmit = function(e) { e.preventDefault(); const email = document.getElementById('email').value; const password = document.getElementById('password').value; loginBtn.innerText = "Opening Club Gates..."; loginBtn.disabled = true; auth.signInWithEmailAndPassword(email, password) .then((userCredential) => { console.log("Login Success!"); // We use replace instead of href for a faster, smoother jump on mobile window.location.replace("dashboard.html"); // Fallback: If it doesn't move in 1 second, try the other method setTimeout(() => { window.location.href = "dashboard.html"; }, 1000); }) .catch((error) => { loginBtn.innerText = "Sign In"; loginBtn.disabled = false; if (error.code === 'auth/user-not-found' || error.code === 'auth/wrong-password') { alert("Invalid email or password. Please try again."); } else { alert("Error: " + error.message); } }); };
